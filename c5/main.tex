\qs{}{
    $$
        \lim_{x \rightarrow \infty} \left( \frac{x^2 + x - 1}{2x^2 - x + 1 } \right)^3
    $$
}

\begin{align}
    \lim_{x \rightarrow \infty}& \left( \frac{x^2 + x - 1}{2x^2 - x + 1 } \right)^3 \\
    &= \left( \frac{\infty}{\infty} \right)^3 \\
    &= \frac{x^2(1 + x^{-1} + x^{-2})}{x^2(2 - x^{-1} + x^{-2})} \\
    &= \left( \frac{1}{2} \right)^3 = \frac{1}{8}
\end{align}
