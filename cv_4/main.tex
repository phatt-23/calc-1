\section{Limita funkce}

\defi{Chování limita exponenciální funkce $f(q) \coloneq q^n$.}{
    \begin{align*}
        \lim q^n 
        \begin{cases}
            +\infty, & \text{if}\ q > 1 \\
            1, & \text{if}\ q = 1 \\
            0, & \text{if}\ q = 0 \\
            0, & \text{if}\ |q| < 1 \\
            \text{DNF}, & \text{if}\ q \le -1  
        \end{cases} \\         
    \end{align*}    
}
