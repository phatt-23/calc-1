\qs{}{
    $$ \lim_{\ntoinf} \frac{3^n + (-3)^n}{3\cdot 6^n} $$
}

\begin{align}
    \lim_{\ntoinf} & \frac{3^n + (-3)^n}{3 \cdot 6^n} \\
    \lim_{\ntoinf} & \frac{6^n \left( \frac{1}{2} - \frac{1}{2} \right)}{6^n \cdot 3} \\
    \lim_{\ntoinf} & \frac{6^n \cdot 0}{6^n \cdot 3} \\
    \lim_{\ntoinf} & \frac{0}{3} = 0
\end{align}
