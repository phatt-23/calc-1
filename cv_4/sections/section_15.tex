\qs{}{
    $$ \lim_{\ntoinf} \frac{1}{n^2} + \frac{2}{n^2} + \ldots + \frac{n - 1}{n^2}$$ 
}

\begin{align}
    \lim_{\ntoinf} & \frac{1 + 2 + \ldots + (n - 1)}{n^2} \\
    \lim_{\ntoinf} & \frac{n(n - 1)}{2n^2} \\
    \lim_{\ntoinf} & \frac{n^2 - n^1}{2n^2} \\
    \lim_{\ntoinf} & \frac{n^2(1 - n^{-1})}{2n^2} \\
    \lim_{\ntoinf} & \frac{1 - \frac{1}{n}}{2} = \frac{1}{2}
\end{align}

\defi{Nekonečná řada sumy inkrementujicích se integrálů}{
    $$ 1 + 2 + \ldots + n &= \frac{n(n + 1)}{2} $$
    $$ \sum_{i = 1}^{n} i = \frac{n(n + 1)}{2}$$
    \begin{align*}
        S &= 1 + 2 + \ldots + (n - 1) + n \\
        2S &= ((n) + (1)) + ((n - 1) + (2)) + \ldots + ((2) + (n - 1)) + ((1) + (n)) \\
        &= (n + 1) + (n + 1) + \ldots + (n + 1) + (n + 1) \\
        &= n(n + 1)
        S = \frac{n(n + 1)}{2}
    \end{align*}
}
