\qs{}{
    $$ \lim_{\ntoinf} \frac{3(n + 2)! - (n + 1)!}{(n + 3)!} $$
}

\begin{align}
    \lim_{\ntoinf} & \frac{(n + 3)! \left( \frac{3}{n + 3} - \frac{1}{(n + 3)(n + 2)} \right)}{(n + 3)!} \\
    \lim_{\ntoinf} & \left( \frac{3}{n + 3} - \frac{1}{(n + 3)(n + 2)} \right) \\
    \lim_{\ntoinf} & \left( \frac{3}{\infty} - \frac{1}{\infty} \right) = 0
\end{align}
