\qs{Vypočtěte limitu}{
    $$ \lim_{\ntoinf}\frac{-2n^3 - 4n + 2}{5n^2 + n - 8} $$
}

\begin{align}
    &\lim_{\ntoinf}\frac{-2n^3 - 4n + 2}{5n^2 + n - 8} \\
    &= \lim_{\ntoinf}\frac{(-2n^3 - 4n + 2)'}{(5n^2 + n - 8)'} \\
    &= \lim_{\ntoinf}\frac{-6n^2 - 4}{10n + 1} \\
    &= \lim_{\ntoinf}\frac{(-6n^2 - 4)'}{(10n + 1)'} \\
    &= \lim_{\ntoinf}\frac{-10n}{10} \\ 
    &= \lim_{\ntoinf}-n = -\infty
\end{align}
