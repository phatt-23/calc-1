}

    \lim_{\ntoinf}\frac{(5n^2 + 9n + 6)'}{(-6n^2 + 3n - 1)'} \\

    \lim_{\ntoinf}\frac{(10n + 9)'}{(-12n + 3)'} \\
    \lim_{\ntoinf}\frac{10}{-12} = -\frac{5}{6} \\
\end{align}
