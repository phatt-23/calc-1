\qs{Vypočtěte limitu}{$$ 
    \lim_{\ntoinf} \left( 1 + \frac{1}{n} \right)^{2n}
$$}

\begin{align}
    \lim_{\ntoinf} \left( 1 + \frac{1}{n} \right)^{2n} \\
    \lim_{\ntoinf} \left( \left( 1 + \frac{1}{n} \right)^n \right)^2 \\
    \lim_{\ntoinf} \left( e \right)^2 = e^2 \\
\end{align}
