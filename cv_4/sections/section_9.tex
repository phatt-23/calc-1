\qs{}{$$ 
    \lim_{\ntoinf} \left( 1 + \frac{1}{n} \right)^{3n + 6}
$$}

\begin{align}
    \lim_{\ntoinf}&\left( 1 + \frac{1}{n} \right)^{3n} 
    \cdot \left( 1 + \frac{1}{n} \right)^{6} \\
    \lim_{\ntoinf}&e^3
    \cdot \left( 1 + \frac{1}{n} \right)^{6} \\
    \lim_{\ntoinf}&e^3
    \cdot 1 \\
\end{align}
