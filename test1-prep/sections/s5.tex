\qs{Určete definiční obor funkce $f$}{
	$$ f(x) = \sqrt{\ln{( x^2 - 4x + 4 )}} $$
}

\begin{align}
	\ln{(x^2 - 4x + 4)} &\geq 0	\\
	x^2 - 4x + 4 &\geq 1 \\
	x^2 - 4x + 3 &\geq 0 \\
	D &= 16 - 4 \cdot 3 = 4 \\
	x_{1,2} &= \frac{4 \pm 2}{2} \\
	x_{1,2} &= \{1, 3\} \\
\end{align}

Definiční obor funkce $f$ je
\[ 
	D_f = \left( -\infty, 1 \right> \cap \left< 3, \infty \right) 
	\rightarrow \mathbb{R} \setminus \left\{ \left( 1,3 \right) \right\}   
\]
