\qs{Vypočítejte limitu funkce}{
	\[ \lim_{x \rightarrow -3} \frac{x^2 - 21x - 72}{x^3 + 27} \]
}

\begin{align}
	\lim_{x \rightarrow -3}& \frac{\left(x^2 - 21x - 72\right)'}{\left(x^3 + 27\right)'} \\
	\lim_{x \rightarrow -3}& \frac{2x - 21}{3x^2} \\
	&= \frac{2(-3) - 21}{3(-3)^2} \\
	&= \frac{-27}{27} = -1
\end{align}
